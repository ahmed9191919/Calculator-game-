using UnityEngine; using UnityEngine.AI; public class SurvivalGameManager : MonoBehaviour { // ===== WORLD GENERATION ===== [Header("World Settings")] public int size = 512; public float heightScale = 20f; public float detailScale = 25f; public Terrain terrain; // ===== PLAYER STATS ===== [Header("Player Stats")] public Transform player; public float health = 100f; public float hunger = 100f; public float hungerDecreaseRate = 0.1f; // ===== BOSS SETTINGS ===== [Header("Boss Settings")] public float bossHealth = 500f; public NavMeshAgent bossAgent; public Transform bossTarget; public GameObject bossDeathDrop; void Start() { // Generate world terrain if (terrain != null) terrain.terrainData = GenerateTerrain(terrain.terrainData); } void Update() { HandlePlayerSurvival(); HandleBossAI(); } // ====== WORLD GENERATION ====== TerrainData GenerateTerrain(TerrainData terrainData) { terrainData.heightmapResolution = size + 1; terrainData.size = new Vector3(size, heightScale, size); terrainData.SetHeights(0, 0, GenerateHeights()); return terrainData; } float[,] GenerateHeights() { float[,] heights = new float[size, size]; for (int x = 0; x < size; x++) { for (int y = 0; y < size; y++) { float xCoord = (float)x / size * detailScale; float yCoord = (float)y / size * detailScale; heights[x, y] = Mathf.PerlinNoise(xCoord, yCoord); } } return heights; } // ====== PLAYER SURVIVAL ====== void HandlePlayerSurvival() { hunger -= hungerDecreaseRate * Time.deltaTime; hunger = Mathf.Clamp(hunger, 0, 100); if (hunger <= 0) health -= 0.2f * Time.deltaTime; if (health <= 0) Debug.Log("Player has died!"); } // ====== BOSS AI ====== void HandleBossAI() { if (bossAgent != null && bossTarget != null) bossAgent.SetDestination(bossTarget.position); } public void DamageBoss(float damage) { bossHealth -= damage; if (bossHealth <= 0) BossDies(); } void BossDies() { if (bossDeathDrop != null) Instantiate(bossDeathDrop, bossAgent.transform.position, Quaternion.identity); Destroy(bossAgent.gameObject); Debug.Log("Boss Defeated!"); } }
